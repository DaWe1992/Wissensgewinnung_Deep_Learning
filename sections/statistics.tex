\section{Grundlagen der Statistik}

\subsection{Grundbegriffe der Wahrscheinlichkeitstheorie}

% Wahrscheinlichkeitsräume
\begin{frame}{Wahrscheinlichkeitsräume}{}
	Wir definieren zunächst, was man unter einem \keyword{Wahrscheinlichkeitsraum} versteht.
	\begin{itemize}
		\item Ein Wahrscheinlichkeitsraum ist ein Tripel $(\Omega, \sA, \Prob)$.
		\item $\Omega$ ist der \keyword{Ereignisraum}, also die Menge aller Elemantarereignisse.
			Beim Werfen eines sechsseitigen Würfels betrachten wir $\Omega \coloneqq \{ 1, 2, 3, 4, 5, 6 \}$.
		\item Mit $\sA$ bezeichnen wir die zugehörige \keyword{Sigma-Algebra}, einer Menge von Teilmengen von $\Omega$. Im einfachsten Fall könnten wir
			$\sA \coloneqq \Pot(\Omega)$ wählen, jedoch kann man zeigen, dass die Potenzmenge im Allgemeinen nicht messbare Mengen enthält,
			denen keine Wahrscheinlichkeit zugeordnet werden kann. Die Sigma-Algebra $\B$ auf $\R$ heißt \keyword{Borel-Sigma-Algebra}.
		\item $\Prob : \sA \rightarrow [0, 1]$ ist ein \keyword{Wahrscheinlichkeitsmaß}, das jedem Element aus $\sA$ eine Wahrscheinlichkeit zuordnet.
	\end{itemize}
\end{frame}


% Zufallsvariablen
\begin{frame}{Zufallsvariablen}{}
	Im Folgenden sei $(\Omega, \sA, \Prob)$ ein Wahrscheinlichkeitsraum und $(\sX, \sB)$ ein messbarer Raum.
	
	\begin{mydef}[def:zufallsvariable]
		Die Abbildung $x : \Omega \rightarrow \sX$ heißt \keyword{Zufallsvariable}, falls
		$$\forall B \in \sB : x^{-1}(B) \coloneqq \{ \omega \in \Omega : x(\omega) \in B \} \in \sA$$ gilt.
		Das Urbild von $B$ unter $x$ muss also eine messbare Menge in $\sA$ sein, der durch $\Prob$ eine Wahrscheinlichkeit zugeordnet werden kann.
		Ferner heißt für jede Zufallsvariable $x : \Omega \rightarrow \sX$ die Funktion
		$$\Prob^x : \sB \rightarrow [0, 1] \qquad\text{mit}\qquad B \mapsto \Prob\bigl( x^{-1}(B) \bigr)$$
		die \keyword{Verteilung} der Zufallsvariablen $x$. $\Prob^x$ ist das \keyword{Bildmaß} von $\Prob$ unter $x$.
	\end{mydef}
\end{frame}


% Zufallsvariablen (Forts.)
\begin{frame}{Zufallsvariablen (Forts.)}{}
	Sei $(\sX, \sB, \Q)$ ein Wahrscheinlichkeitsraum. Man sagt, die Zufallsgröße $x : \Omega \rightarrow \sX$ ist
	\textit{verteilt gemäß $\Q$} oder kurz \textit{$\Q$-verteilt}, falls es einen Wahrscheinlichkeitsraum $(\Omega, \sA, \Prob)$ gibt,
	sodass $\Prob^x = \Q$ gilt.
	
	\textbf{Beispiele:}
	\begin{itemize}
		\item $x$ ist $N_{\mu, \sigma^2}$-verteilt genau dann, wenn es einen Wahrscheinlichkeitsraum $(\Omega, \sA, \Prob)$ gibt, sodass
			$\forall B \in \B : \Prob(x \in B) = N_{\mu, \sigma^2}(B)$. Wir schreiben kurz $x \sim N_{\mu, \sigma^2}$.
		\item 
	\end{itemize}
\end{frame}


% Wahrscheinlichkeitsdichten
\begin{frame}{Wahrscheinlichkeitsdichten}{}
	In der Regel wollen wir mit Dichten rechnen:
	\begin{mydef}[def:dichte]
		Es seien $(\Omega, \sA, \Prob)$ und $(\sX, \sB, \Prob^x)$ Wahrscheinlichkeitsräume. Dabei ist $\Prob^x$ die Verteilung der Zufallsvariablen
		$x : \Omega \rightarrow \sX$.
		Wir nennen eine nichtnegative Funktion $$p_x : \R \rightarrow [0, 1]$$ eine \keyword{(Wahrscheinlichkeits-)Dichte} von $\Prob^x$, falls
		$$\Prob^x(B) = \int_B p_x(\xi)\,\diff\xi \qquad \forall B \in \sB.$$ Die Wahrscheinlichkeiten der Mengen $B \in \sB$ müssen sich also allesamt als
		Integral der Dichte $p_x$ darstellen lassen können.
	\end{mydef}
\end{frame}


% Erwartungswert von Zufallsvariablen
\begin{frame}{Erwartungswert von Zufallsvariablen}{}
	\begin{mydef}[def:erwartungswert]
		Sei $x \in \R$ eine Zufallsvariable. Wir definieren den \keyword{Erwartungswert} (erstes Moment) von $x$ als
		$$\E[x] \coloneqq \int_{\R} \xi \cdot p_x(\xi)\,\diff\xi.$$
	\end{mydef}

	Aus der Linearität des Integrals folgt die des Erwartungswerts, d.\,h. für zwei Zufallsvariablen $x$ und $y$ und eine beliebige aber feste Zahl
	$\alpha \in \R$ (keine Zufallsvariable!) gelten $$\E[x + y] = \E[x] + \E[y] \qquad\text{und}\qquad \E[\alpha \cdot x] = \alpha \cdot \E[x].$$
	Wir definieren den Erwartungswert einer vektorwertigen Zufallsvariablen $\bx = (x_1, x_2, \ldots, x_n)^\top$ als
	$$\E[\bx] \coloneqq \bigl( \E[x_1], \E[x_2], \ldots, \E[x_n] \bigr)^\top.$$
\end{frame}


% Varianz von Zufallsvariablen
\begin{frame}{Varianz von Zufallsvariablen}{}
	\begin{mydef}[def:varianz]
		Die Varianz einer Zufallsvariablen $x \in \R$ ist definiert als die erwartete quadratische Abweichung vom Erwartungswert
		(zweites zentriertes Moment), also $$\V[x] \coloneqq \E\bigl[ (x - \E[x])^2 \bigr] = \int_\R \bigl( \xi - \E[x] \bigr)^2 \cdot p_x(\xi)\,\diff\xi.$$
	\end{mydef}

	Unter Ausnutzung der Linearität des Erwartungswerts können wir die Varianz folgendermaßen umschreiben:
	\begin{align*}
		\V[x]
			&= \E\bigl[ (x - \E[x])^2 \bigr] = \E\bigl[ x^2 - 2x \cdot \E[x] + \bigl( \E[x] \bigr)^2 \bigr]
	\\[2mm]
			&= \E[x^2] - 2 \bigl( \E[x] \bigr)^2 + \bigl( \E[x] \bigr)^2
	\\[2mm]
			&= \E[x^2] - \bigl( \E[x] \bigr)^2
	\end{align*}
\end{frame}


% Varianz von Zufallsvariablen (Forts.)
\begin{frame}{Varianz von Zufallsvariablen (Forts.)}{}
	Ein paar Bemerkungen. Seien dazu $x$ und $y$ reelle Zufallsvariablen und $\alpha \in \R$ beliebig aber fest.
	\begin{itemize}
		\item Die Varianz ist \textbf{nicht} linear:
		\begin{itemize}
			\item Es gilt $\V[x + y] = \V[x] + \V[y]$ nur, wenn $x$ und $y$ unkorreliert sind (Satz von Bienaymé).
			\item $\V[\alpha \cdot x] = \alpha^2 \cdot \V[x]$
		\end{itemize}
		\item Eine Verschiebung verändert die Varianz nicht, also $\V[x + \alpha] = \V[x]$
	\end{itemize}
\end{frame}

% Stochastische Unabhängigkeit
\begin{frame}{Stochastische Unabhängigkeit}{}
	Mit Dichten können wir auch die \keyword{stochastische Unabhängigkeit} von Zufallsvariablen definieren:
	
	\begin{mydef}[def:stoch-unab]
		Es seien $x \in \R$ und $y \in \R$ reelle Zufallsvariablen mit den Verteilungen $\Prob^x, \Prob^y$ und den Dichten $p_x$ bzw. $p_y$.

		$x$ und $y$ heißen \keyword{stochastisch unabhängig} genau dann wenn $p_{x, y} \coloneqq p_x \cdot p_y$ eine Produktdichte der Verteilung
		$\Prob^{x, y}$ ist, d.\,h. die gemeinsame Dichte von $x$ und $y$ ergibt sich als Produkt der Einzeldichten. 
	\end{mydef}

	\textbf{Bemerkung:} Stochastisch unabhängige Zufallsvariablen sind auch unkorreliert. Die Umkehrung gilt im Allgemeinen nicht.
	Somit ist die stochastische Unabhängigkeit eine stärkere Forderung als Unkorreliertheit.
\end{frame}

\subsection{Transformationssatz}

% Transformationssatz für Lebsgue-Dichten
\begin{frame}{Transformationssatz für Lebesgue-Dichten}{}
	\begin{mytheo}[th:transformationssatz]
		Es seien $\bx$ und $\by$ $n$-dimensionale Zufallsvektoren mit dem Zusammenhang $$\by = \bT(\bx).$$
		Die Abbildung $\bT : \R^n \rightarrow \R^n$ sei \textit{bijektiv}, sodass die Umkehrabbildung $\bx = \bT^{-1}(\by)$ existiert und eindeutig ist.
		Dann gilt für die Dichte $p_{\by}$ der transformierten Zufallsvariablen $\by$
		\begin{equation*}
			p_{\by}(\by) = p_{\bx}\bigl( \bT^{-1}(\by) \bigr) \cdot \bigl\vert \det J\bT^{-1}(\by) \bigr\vert,
		\end{equation*}
		wobei $p_{\bx}$ die Dichte von $\bx$ und $J \bT^{-1}$ die Jacobi-Matrix der Abbildung $\bT^{-1}$ bezeichnet.
	\end{mytheo}
	
	\begin{myproof}
		Für den Beweis werden Argumente aus der Maß- und Integrationstheorie benötigt, deren Behandlung den Rahmen dieses Kurses übersteigt.
	\end{myproof}
\end{frame}


% Beispiel I
\begin{frame}{Beispiel I}{}
	Es sei $x \sim Exp_2$ eine reellwertige exponentialverteilte Zufallsvariable mit Ankunftsrate $\lambda = 2$. \\
	Gesucht ist die Dichte der Zufallsvariablen $y \coloneqq 5x + 2$.
	
	\textbf{Lösung:} Es ist $T(x) \coloneqq 5 x + 2$ eine bijektive Transformation mit der Umkehrabbildung $x = T^{-1}(y) = \frac{1}{5}(y - 2)$.
	Zusammen mit der Dichte $p_x(x) = 2 \cdot e^{-2 x} \cdot \indicator_{[0, \infty)}(x)$ der $Exp_2$-Verteilung erhalten wir für die Dichte $p_y(y)$
	der transformierten Zufallsvariablen $y$
	\begin{align*}
		p_y(y)
			&= p_x\bigl( T^{-1}(y) \bigr) \cdot \bigl( T^{-1} \bigr)'(y)
	\\[2mm]
			&= 2 \cdot e^{-2 \left( \frac{y}{5} - \frac{2}{5} \right)} \cdot \indicator_{[0, \infty)}\left( \frac{1}{5} y - \frac{2}{5} \right) \cdot \frac{1}{5}
	\\[2mm]
			&= \frac{2}{5} \cdot e^{\frac{4}{5}} \cdot e^{-\frac{2 y}{5}} \cdot \indicator_{[2, \infty)}(y).
	\end{align*}
\end{frame}


% Beispiel II
\begin{frame}{Beispiel II}{}
	Der Zufallsvektor $\bx \in \R^n$ habe eine beliebige Lebesgue-Dichte $p_{\bx}$. Die Transformation $\bT$ sei linear und bijektiv,
	also von der Form $\by = \bT(\bx) = \bA \bx + \bb$. Insbesondere ist $\bA$ in diesem Fall invertierbar.
	Dann existiert die Umkehrfunktion und ist gegeben durch $\bx = \bT^{-1}(\by) = \bA^{-1} (\by - \bb)$.
	Als Jacobi-Matrix erhält man $J\bT^{-1}(\by) = \bA^{-1}$.
	
	Die Anwendung der Transformationsformel \ref{th:transformationssatz} ergibt daher für die Dichte $p_{\by}$ von $\by$ den Ausdruck
	$$p_{\by}(\by) = \frac{1}{\vert \det \bA \vert} \cdot p_{\bx}\bigl( \bA^{-1} (\by - \bb) \bigr),$$ wobei wir den Zusammenhang
	$\det\bigl( \bA^{-1} \bigr) = \frac{1}{\det \bA}$ benutzt haben.
\end{frame}


\subsection{Multivariate Normalverteilung}

% Multivariate Normalverteilung
\begin{frame}{Multivariate Normalverteilung}{}
	Die multivariate Normalverteilung spielt im weiteren Verlauf eine zentrale Rolle:
	\begin{mytheo}[th:multivar-stdnv]
		Es sei $\bx \coloneqq (x_1, x_2, \ldots, x_n)^\top \in \R^n$ ein Zufallsvektor, dessen Komponenten $x_i \in \R$ unabhängig und
		identisch standardnormalverteilt sind, d.\,h. $x_i \sim N_{0, 1}\,\forall i \in \{ 1, \ldots, n \}$.
		
		Die Dichte der \keyword{multivariaten Standardnormalverteilung} ist dann gegeben durch
		$$p_{\bx}(\bx) \coloneqq \frac{1}{(2 \pi)^{\frac{n}{2}}} \cdot \exp\left( -\frac{\Vert \bx \Vert^2}{2} \right).$$
	\end{mytheo}

	\begin{myproof}
		Wegen der Unabhängigkeit der Komponenten gilt für die Verbunddichte
		\begin{equation*}
			p_{\bx}(\bx)
				= \prod_{i=1}^n p_{x_i}(x_i)
				= \prod_{i=1}^n \frac{1}{\sqrt{2 \pi}} \cdot \exp\left( -\frac{x_i^2}{2} \right)
				= \frac{1}{(2 \pi)^{\frac{n}{2}}} \cdot \exp\left( -\frac{\sum_{i=1}^n x_i^2}{2} \right)
				= \frac{1}{(2 \pi)^{\frac{n}{2}}} \cdot \exp\left( -\frac{\bx^\top \bx}{2} \right)
				= \frac{1}{(2 \pi)^{\frac{n}{2}}} \cdot \exp\left( -\frac{\Vert \bx \Vert^2}{2} \right).
		\end{equation*}
	\end{myproof}
\end{frame}


% Multivariate Normalverteilung (Forts.)
\begin{frame}{Multivariate Normalverteilung (Forts.)}{}
	Für die multivariate Standardnormalverteilung gilt $\E[\bx] = \0$ und $\D[\bx] = \bI$.

	Wir verallgemeinern dies auf beliebige Erwartungswertvektoren $\bmu \in \R^n$ und Streuungsmatrizen $\bSigma \in \R^{n \times n}$.
	Wir fordern lediglich, dass die Streuungsmatrix $\bSigma$ \textit{positiv definit} ist, dass also
	$\bz^\top \bSigma \bz > 0\, \forall \bz \in \R^n \backslash \{ \0 \}$ gilt.

	\begin{enumerate}
		\item Aufgrund der positiven Definitheit von $\bSigma$ gibt es eine untere Dreiecksmatrix $\bA$ mit $\bSigma = \bA \bA^\top$
			(Stichwort: \keyword{Cholesky-Zerlegung}).
		\item Wir betrachten die Transformation $\by \coloneqq \bA \bx + \bmu$. Es gilt
			$$\E[\by] = \E[\bA \bx + \bmu] = \bA \E[\bx] + \bmu = \bmu \qquad\text{und}\qquad
			\D[\by] = \D[\bA \bx + \bmu] = \bA \D[\bx] \bA^\top = \bA \bA^\top = \bSigma.$$
		\item Nun wenden wir die Transformationsformel \ref{th:transformationssatz} an.
	\end{enumerate}
\end{frame}


% Multivariate Normalverteilung (Forts.)
\begin{frame}{Multivariate Normalverteilung (Forts.)}{}
	Es ist $\bT^{-1}(\bx) = \bA^{-1} (\by - \bmu)$. Mit obigem Beispiel II zum Transformationssatz erhalten wir
	$$\bigl\vert \det J \bT^{-1}(\by) \bigr\vert = \bigl\vert \det \bA^{-1} \bigr\vert = \frac{1}{\vert \det \bA \vert}.$$ Außerdem gilt mit dem
	Determinantenmultiplikationssatz\footnote{Für zwei $n \times n$ Matrizen $\bA$ und $\bB$ gilt $\det(\bA \bB) = \det \bA \cdot \det \bB$.}
	die Identität $\det \bSigma = \det\bigl( \bA \bA^\top \bigr) = \det \bA \cdot \det \bA = \det^2 \bA$.
	Durch Ziehen der Wurzel auf beiden Seiten erhalten wir also $\vert \det \bA \vert = (\det \bSigma)^{\frac{1}{2}}$. 
	Die Dichte $p_{\by}$ nimmt damit die Form
	$$p_{\by}(\by) = \frac{1}{(\det \bSigma)^\frac{1}{2}} \cdot p_{\bx}\bigl( \bA^{-1} (\by - \bmu) \bigr)$$ an,
	wobei $p_{\bx}$ die Dichte der Standardnormalverteilung aus Satz \ref{th:multivar-stdnv} ist. Wir setzen dort
	$\bx \coloneqq \bA^{-1} (\by - \bmu)$, sodass
	$\Vert \bx \Vert^2 = (\by - \bmu)^\top \bigl(\bA^{-1})^\top \bA^{-1} (\by - \bmu) = (\by - \bmu)^\top \bSigma^{-1} (\by - \bmu)$ gilt. 
\end{frame}


% Multivariate Normalverteilung (Forts.)
\begin{frame}{Multivariate Normalverteilung (Forts.)}{}
	Wir fassen zusammen:
	
	\begin{mydef}[def:multivar-nv]
		Es sei $\bmu \in \R^n$ und $\bSigma \in \R^{n \times n}$ eine \textit{positiv definite} Matrix.
		
		Die Dichte der multivariaten Normalverteilung $N_{\bmu, \bSigma}$ ist gegeben durch
		$$p_{\bx}(\bx) \coloneqq \frac{1}{(2 \pi)^{\frac{n}{2}} \cdot (\det \bSigma)^{\frac{1}{2}}} \cdot \exp\left(
			-\frac{1}{2} (\bx - \bmu)^\top \bSigma^{-1} (\bx - \bmu) \right).$$
	\end{mydef}
	
	\textbf{Bemerkung:} Die positive Definitheit der Matrix $\bSigma$ impliziert deren Invertierbarkeit.
\end{frame}

% Partitionierte Normalverteilung
\begin{frame}{Partitionierte Normalverteilung}{}
	\begin{mydef}[def:partitionierte-nv]
		Es seien $\bx \in \R^d$ und $\by \in \R^{(n - d)}$ Zufallsvektoren.
		Wir fassen diese zu einem neuen Zufallsvektor $\bz \coloneqq \bigl( \bx^\top, \by^\top \bigr)^\top \in \R^n$
		mit $\bz \sim N_{\bmu, \bSigma}$ zusammen. Hierbei seien
		$$\bmu \coloneqq \begin{pmatrix} \bmu_{\bx} \\ \bmu_{\by} \end{pmatrix} \qquad\text{und}\qquad \bSigma \coloneqq
		\begin{pmatrix} \bSigma_{\bx\bx} & \bSigma_{\bx\by} \\ \bSigma_{\by\bx} & \bSigma_{\by\by} \end{pmatrix}.$$
		Wir nennen dieses Modell \keyword{partitionierte Normalverteilung}.
	\end{mydef}
	
	\textbf{Bemerkungen:}
	\begin{itemize}
		\item Aus der Symmetrie von $\bSigma$ folgt die von $\bSigma_{\bx\bx}$ und $\bSigma_{\by\by}$.
			Außerdem gilt $\bSigma_{\bx\by} = \bSigma_{\by\bx}^\top$.
		\item Oft ist es einfacher mit der Inversen der Kovarianzmatrix zu arbeiten. Wir definieren daher $\bLambda \coloneqq \bSigma^{-1}$.
			Man nennt diese Matrix \keyword{Präzisionsmatrix}.
	\end{itemize}
\end{frame}


% Bedingte Normalverteilung
\begin{frame}{Bedingte Normalverteilung}{}
	\begin{mytheo}[th:bedingte-nv]
		Wir betrachten die partitionierte Normalverteilung aus Definition \ref{def:partitionierte-nv}. Dann ist die Verteilung von $\bx \mid \by$
		ebenfalls normalverteilt mit den Parametern
		\begin{align*}
			\bmu_{\bx \mid \by}
				&\coloneqq \bmu_{\bx} + \bSigma_{\bx\by} \bSigma_{\by\by}^{-1}(\by - \mu_{\by})
		\\[2mm]
			\bSigma_{\bx \mid \by}
				&\coloneqq \bSigma_{\bx\bx} - \bSigma_{\bx\by} \bSigma_{\by\by}^{-1} \bSigma_{\by\bx}
		\end{align*}
		Die Dichte dieser Verteilung bezeichnen wir mit $p_{\bx \mid \by}$.
	\end{mytheo}

	\begin{myproof}
		Einen Beweis dieser Aussage finden Sie in Bishop (2006) auf den Seiten 85--87.
	\end{myproof}
\end{frame}

% Normale Randverteilung
\begin{frame}{Normale Randverteilung}
	\begin{mytheo}[th:rand-nv]
		Wieder betrachten wir die partitionierte Normalverteilung aus Definition \ref{def:partitionierte-nv}. Dann sind auch die Randverteilungen von $\bx$ und
		$\by$ normalverteilt. Es gilt
		\begin{align*}
			\bx
				&\sim N_{\bmu_{\bx}, \bSigma_{\bx\bx}}
		\\[2mm]
			\by
				&\sim N_{\bmu_{\by}, \bSigma_{\by\by}}
		\end{align*}
		Die Dichten der Randverteilung bezeichnen wir mit $p_{\bx}$ und $p_{\by}$.
	\end{mytheo}

	\begin{myproof}
		Man erhält dieses Ergebnis durch \keyword{Marginalisierung} der Verbunddichte, d.\,h. wir integrieren über alle Variablen, die uns nicht interessieren.
		Für die Randverteilung von $\bx$ gilt zum Beispiel $$p_{\bx}(\bx) = \int p_{\bx, \by}(\bx, \by)\,\diff \by.$$
		Einen ausführlichen Beweis finden Sie in Bishop (2006) auf den Seiten 88--89.
	\end{myproof}
\end{frame}

\subsection{Maximum-Likelihood Schätzer}

% Maximum-Likelihood Methode
\begin{frame}{Maximum-Likelihood Methode}{}
	\begin{itemize}
		\item Im Folgenden sei ein Beobachtungsvektor $\bx \coloneqq \bigl( x_1, x_2, \ldots, x_N \bigr)^\top$ bestehend aus $N$ \textit{unabhängig und identisch verteilten} (i.\,i.\,d.)
			Datenpunkten gegeben.
		\item Wir möchten Aussagen über die Parameter $\btheta \coloneqq \bigl( \theta_1, \theta_2, \ldots, \theta_M \bigr)^\top$ der zugrundeliegenden Verteilung machen.
		\item Dies erledigen wir durch eine Punktschätzung des Parametervektors $\btheta$. Den Schätzer für $\btheta$ bezeichnen wir mit $\bthetaest$.
		\item Eine der am häufigsten angewendete Methode ist die \keyword{Maximum-Likelihood Methode}.
	\end{itemize}
\end{frame}


% (Log-)Likelihood Funktion
\begin{frame}{(Log-)Likelihood Funktion}{}
	Der Maximum-Likelihood Schätzer $\bthetaestML$ für den Parametervektor $\btheta$ maximiert die Likelihood Funktion.
	
	\begin{mydef}[def:likelihood-funktion]
		Es sei $p(\bx; \btheta)$ die Dichte eines Zufallsvektors $\bx \in \R^n$, welche durch den Parametervektor $\btheta \in \Theta$ parametrisiert wird.
		Die \keyword{Likelihood Funktion} $L : \Theta \rightarrow \R$ ist dann definiert als $$L(\btheta) \coloneqq p(\bx; \btheta) = p(x_1, x_2, \ldots, x_N; \btheta).$$
		Häufig ist es einfacher, die \keyword{Log-Likelihood Funktion} $$l(\btheta) \coloneqq \ln p(\bx; \btheta)$$ zu betrachten.
		Es ist $\bthetaestML \coloneqq \argmax_{\btheta} l(\btheta).$
	\end{mydef}

	\textbf{Bemerkung:} Der Schätzer $\bthetaestML$ maximiert sowohl $L$ als auch $l$. \textbf{(Warum?)} 
\end{frame}


% (Log-)Likelihood Funktion (Forts.)
\begin{frame}{(Log-)Likelihood Funktion (Forts.)}{}
	\begin{itemize}
		\item Im Allgemeinen kann es schwer sein, die Liklihood Funktion zu bestimmen.
		\item Die Annahme der Unabhängigkeit der Beobachtungen vereinfacht dies maßgeblich, denn
		\begin{align*}
			L(\btheta)
				&= p(x_1, x_2, \ldots, x_N; \btheta)
					= p(x_1; \btheta) \cdot p(x_2; \btheta) \cdot \ldots \cdot p(x_N; \btheta)
			\\[2mm]
				&= \prod_{n=1}^N p(x_n; \btheta).
		\end{align*}
		\item Für die Log-Likelihood Funktion ergibt sich durch Anwendung der Logarithmengesetze $$l(\btheta) = \ln \prod_{n=1}^N p(x_n; \btheta) = \sum_{n=1}^N \ln p(x_n; \btheta).$$
	\end{itemize}
\end{frame}


% Maximum-Likelihood Gleichung
\begin{frame}{Maximum-Likelihood Gleichung}{}
	\begin{itemize}
		\item Falls die Likelihood Funktion differenzierbar ist, so erfüllt der Maximum-Likelihood Schätzer $\bthetaestML$ zwingend die \keyword{Likelihood Gleichung}
			$$\frac{\partial}{\partial \btheta} \ln p(\bx; \btheta)\,\bigg\vert_{\btheta = \bthetaestML} = \0.$$
		\item Diese sagt aus, dass $\bthetaestML$ ein stationärer Punkt der (Log-)Likelihood Funktion sein muss.
		\item Es handelt sich dabei um ein System von $M$, im Allgemeinen nicht-linearen, Gleichungen der Form
			$$\frac{\partial}{\partial \theta_j} \ln p(\bx; \btheta)\,\bigg\vert_{\btheta = \bthetaestML} = 0 \qquad j = 1, 2, \ldots, M.$$
		\item \textbf{Achtung:} Es muss noch überprüft werden, ob $\bthetaestML$ tatsächlich ein Maximum darstellt.
	\end{itemize}
\end{frame}


% Beispiel: Binomialverteilung
\begin{frame}{Beispiel: Binomialverteilung}{}
	Es sei eine binomialverteilte Zufallsvariable $x$ (Anzahl Erfolge) mit Stichprobenumfang $N$ gegeben.
	Wir möchten den ML Schätzer $\thetaestML$ für die unbekannte Erfolgswahrscheinlichkeit $\theta \in (0, 1)$ bestimmen.
	
	\begin{enumerate}
		\item Die Likelihood Funktion lautet $L(\theta) = {N \choose x} \cdot \theta^x \cdot (1 - \theta)^{N - x}.$
		\item Durch Anwendung des Logarithmus erhalten wir die Log-Likelihood Funktion $$l(\theta) = \ln{N \choose x} + x \cdot \ln \theta + (N - x) \cdot \ln(1 - \theta).$$
		\item Durch Ableiten nach $\theta$ und Nullsetzen erhalten wir
			$$\frac{\diff}{\diff \theta} l(\theta) = \frac{x}{\theta} + \frac{N - x}{1 - \theta} \overset{!}{=} 0 \quad\Longleftrightarrow\quad \thetaestML = \frac{x}{N}.$$
		\item Es ist $\frac{\diff^2}{\diff \theta^2} l(\theta) = -\frac{x}{\theta^2} - \frac{N - x}{(1 - \theta)^2} < 0\,\forall \theta \in (0, 1)$, sodass $\thetaestML$ ein Maximum ist.
	\end{enumerate}
\end{frame}


% Beispiel: Normalverteilung
\begin{frame}{Beispiel: Normalverteilung}{}
	\begin{mytheo}[th:mle-nv]
		\newline Es seien $\bx = (x_1, x_2, \ldots, x_N)^\top$ unabhängig und identisch normalverteilte Beobachtungen mit unbekanntem Erwartungswert $\mu \in \R$
		und unbekannter Varianz $\sigma^2 > 0$.
		Der Parametervektor $\btheta$ ist damit $\btheta \coloneqq \bigl( \mu, \sigma^2 \bigr)^\top$.
		
		Die Maximum-Likelihood Schätzer $\hat{\mu}_{ML}$ und $\hat{\sigma}^2_{ML}$ sind gegeben durch
		$$\hat{\mu}_{ML} = \frac{1}{N} \sum_{n=1}^n x_n \qquad\text{und}\qquad \hat{\sigma}^2_{ML} = \frac{1}{N} \sum_{n=1}^N \bigl( x_n - \hat{\mu}_{ML} \bigr)^2.$$
	\end{mytheo}
\end{frame}


% Beweis
\begin{frame}{}{}
	\begin{myproof}
		Die gemeinsame (Lebesgue-)Dichte der Produktverteilung lautet aufgrund der Unabhängigkeit der Beobachtungen
		$$p\bigl( \bx; \mu, \sigma^2 \bigr) = \prod_{n=1}^N \frac{1}{\sqrt{2 \pi} \sigma} \cdot \exp\left( -\frac{(x_n - \mu)^2}{2 \sigma^2} \right)
		= \left( \frac{1}{2 \pi} \right)^{\frac{N}{2}} \frac{1}{\sigma^N} \cdot \exp\left( -\frac{\sum_{n=1}^N (x_i - \mu)^2}{2 \sigma^2} \right).$$ Aufgrund der auftretenden
		Exponentialverteilung bietet sich die Betrachtung der Log-Likelihood Funktion an. Wir erhalten
		\begin{equation*}
			l\bigl( \mu, \sigma^2 \bigr)
				= \ln\bigl( p\bigl( \bx; \mu, \sigma^2 \bigr) \bigr)
				= -\frac{N}{2} \cdot \ln(2 \pi) - N \ln \sigma - \frac{\sum_{n=1}^N (x_n - \mu)^2}{2 \sigma^2}.
		\end{equation*}
		Nun berechnen wir die partiellen Ableitungen von $l$ nach $\mu$ und $\sigma^2$ und erhalten
		$\frac{\partial}{\partial \mu} l\bigl( \mu, \sigma^2 \bigr) = \frac{1}{\sigma^2} \sum_{n=1}^N (x_n - \mu)$ beziehungsweise
		$\frac{\partial}{\partial \sigma^2} l\bigl( \mu, \sigma^2 \bigr) = -\frac{N}{2 \sigma^2} + \frac{1}{2 \sigma^4} \sum_{n=1}^N (x_n - \mu)$. Durch Nullsetzen und auflösen
		erhalten wir die im Satz angegebenen Maximum-Likelihood Schätzer $\hat{\mu}_{ML}$ und $\hat{\sigma}^2_{ML}$.
		
		Es bleibt noch zu zeigen, dass es sich tatsächlich um ein Maximum handelt. Man rechnet nach,
		dass die Hesse-Matrix von $l\bigl( \mu, \sigma^2 \bigr)$ and der Stelle $(\hat{\mu}_{ML}, \hat{\sigma}^2_{ML})$ gegeben ist durch
		$$\frac{\partial^2}{\partial \btheta \partial \btheta} l\bigl( \mu, \sigma^2 \bigr) =
		\begin{pmatrix} -\frac{N}{\hat{\sigma}^2_{ML}} & 0 \\ 0 & -\frac{N}{\hat{\sigma}^4_{ML}} \end{pmatrix}.$$ Diese Matrix ist
		negativ definit, weshalb wir tatsächlich ein Maximum gefunden haben.
	\end{myproof}
\end{frame}

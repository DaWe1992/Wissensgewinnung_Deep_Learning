\section{Expectation-Maximization (EM) Algorithmus}

\divider

\subsection{Problemstellung}

% Problemstellung
\begin{frame}{Problemstellung}{}
	\begin{itemize}
		\item Wir betrachten ein statistisches Modell mit Daten $\bX = \bigl( \bx^1, \bx^2, \ldots, \bx^N \bigr)^\top$, \keyword{latenten Variablen}
			$\bZ = \bigl( \bz^1, \bz^2, \ldots, \bz^N \bigr)^\top$ und Modellparametern $\btheta$.
		\item Unser Ziel ist die Maximierung der Log-Likelihood $\ln p(\bX; \btheta)$.
		\item Da wir die latenten Variablen nicht beobachten, kûÑnnten wir û¥ber sie summieren:
		\begin{equation}
			\ln p(\bX; \btheta) = \ln \sum_{\bZ} p(\bX, \bZ; \btheta)
		\end{equation}
		\item In der Regel ist eine direkte LûÑsung hiervon jedoch zu kompliziert.
		\item Ein LûÑsungsansatz ist der \keyword{Expectation-Maximization (EM) Algorithmus}, welcher das Problem iterativ in zwei Phasen lûÑst.
		\item Mit $\{ \bX, \bZ \}$ bezeichnen wir den \textit{vollstûÊndigen} Datensatz, $\bX$ alleine nennen wir \textit{unvollstûÊndig}.
	\end{itemize}
\end{frame}


\subsection{Ungleichung von Jensen}

% Ungleichung von Jensen
\begin{frame}{Ungleichung von Jensen}{}
	Fû¥r unsere weiteren Betrachtungen benûÑtigen wir die folgende Ungleichung:
	\begin{mytheo}[th:jensen]
		Es sei $X$ eine Zufallsvariable mit endlichem ersten Moment (daraus folgt die Existenz des Erwartungswerts) und $f$ eine
		\textit{konvexe} Funktion. Dann gilt die \keyword{Ungleichung von Jensen}
		\begin{equation*}
			f\bigl( \E[X] \bigr) \le \E\bigl[ f(X) \bigr].
		\end{equation*}
	\end{mytheo}
	
	\textbf{Bemerkungen:}
	\begin{itemize}
		\item Fû¥r konkave Funktionen $f$ gilt die Ungleichung in die andere Richtung: $f\bigl( \E[X] \bigr) \ge \E\bigl[ f(X) \bigr]$.
		\item Wenn $f$  \textit{strikt konvex} bzw. \textit{strikt konkav} ist, so gilt Gleichheit genau dann, wenn $X$ konstant (also nicht zufûÊllig) ist.
	\end{itemize}
\end{frame}


\subsection{Bestimmung der unteren Schranke}

% Bestimmung der unteren Schranke
\begin{frame}{Bestimmung der unteren Schranke}{}
	\begin{itemize}
		\item Entscheidend ist, dass die Optimierung von $p(\bX, \bZ; \btheta)$ oftmals einfacher ist als die von $p(\bX; \btheta)$.
		\item Wir konstruieren daher mittels $p(\bX, \bZ; \btheta)$ eine untere Schranke $\sL$ fû¥r $p(\bX; \btheta)$:
		\begin{align*}
			\ln p(\bX; \btheta)
				&= \ln \sum_{\bZ} p(\bX, \bZ; \btheta)
					= \ln \sum_{\bZ} q(\bZ) \cdot \frac{p(\bX, \bZ; \btheta)}{q(\bZ)}
		\\[2mm]
				&= \ln \E_{\bZ \sim q(\bZ)}\left[ \frac{p(\bX, \bZ; \btheta)}{q(\bZ)} \right]
		\\[2mm]
				&\ge \E_{\bZ \sim q(\bZ)} \left[ \ln \frac{p(\bX, \bZ; \btheta)}{q(\bZ)} \right] \eqqcolon \sL(q, \btheta)
		\end{align*}
		\item Fû¥r die AbschûÊtzung im letzten Schritt haben wir die \keyword{Jensensche Ungleichung}, Satz \ref{th:jensen}, mit
			$f \coloneqq \ln$ (der Logarithmus ist strikt konkav) und $X \coloneqq \frac{p(\bX, \bZ; \btheta)}{q(\bZ)}$ benutzt.
	\end{itemize}
\end{frame}


% Bestimmung der unteren Schranke (Forts.)
\begin{frame}{Bestimmung der unteren Schranke (Forts.)}{}
	\begin{itemize}
		\item Die hergeleitete Ungleichung lûÊsst sich mit den Logarithmengesetzen umschreiben zu
		\begin{equation}
			\label{eq:em-untere-schranke}
			\ln p(\bX; \btheta) \ge \underbrace{
				\E_{\bZ \sim q(\bZ)}\left[ \ln p(\bX, \bZ; \btheta) \right]}_{\text{Erw. vollst. Log-Likelihood}} - 
			\underbrace{\E_{\bZ \sim q(\bZ)}\left[ \ln q(\bZ) \right]}_{\text{Entropie von $q$}}.
		\end{equation}
		\item Die untere Schranke hûÊngt von der Dichte $q$ ab, die wir noch bestimmen mû¥sssen.
		\item Wir mûÑchten $q$ so wûÊhlen, dass $\sL(q, \btheta) \overset{!}{=} \ln p(\bX; \btheta)$ fû¥r das aktuelle $\btheta$ gilt.
		\item Die Ungleichung von Jensen (Satz \ref{th:jensen}) gilt mit Gleichheit, falls
		\begin{equation*}
			\frac{p(\bX, \bZ; \btheta)}{q(\bZ)} = \text{const}.
		\end{equation*}
		\item Wir mû¥ssen also $q(\bZ) \propto p(\bX, \bZ; \btheta)$ wûÊhlen!
	\end{itemize}
\end{frame}


% Bestimmung der unteren Schranke (Forts.)
\begin{frame}{Bestimmung der unteren Schranke (Forts.)}{}
	\begin{itemize}
		\item Daraus erhalten wir
		\begin{align*}
			q(\bZ)
				&= \frac{p(\bX, \bZ; \btheta)}{\sum_{\bZtilde} p(\bX, \bZtilde; \btheta)}
					= \frac{p(\bX, \bZ; \btheta)}{p(\bX; \btheta)}
		\\[2mm]
				&= p(\bZ \mid \bX; \btheta).
		\end{align*}
		\item Dies setzen wir in (\ref{eq:em-untere-schranke}) ein und erhalten die IdentitûÊt
		\begin{equation*}
			\ln p(\bX; \btheta) = \E_{\bZ \sim p(\bZ \mid \bX; \btheta)}\left[ \ln p(\bX, \bZ; \btheta) \right]
				- \E_{\bZ \sim p(\bZ \mid \bX; \btheta)}\left[ \ln p(\bZ \mid \bX; \btheta) \right].
		\end{equation*}
	\end{itemize}
\end{frame}


\subsection{Formulierung des EM-Algorithmus}

% Formulierung des EM-Algorithmus
\begin{frame}{Formulierung des EM-Algorithmus}{}
	\begin{itemize}
		\item Zu jedem Zeitpunkt $t$ haben wir eine SchûÊtzung der Parameter $\btheta^t$, die der Algorithmus in der darauffolgenden Iteration verbessert.
		\item Fû¥r die Formulierung des Algorithmus fû¥hren wir einen Zeitindex ein
		\begin{align*}
			\ln p(\bX; \btheta)
				&= \E_{\bZ \sim p(\bZ \mid \bX; \btheta^t)}\left[ \ln p(\bX, \bZ; \btheta) \right]
					- \E_{\bZ \sim p(\bZ \mid \bX; \btheta^t)}\left[ \ln p(\bZ \mid \bX; \btheta) \right]
		\\[2mm]
				&= Q(\btheta; \btheta^t) + H(\btheta; \btheta^t),
		\end{align*}
			wobei $Q(\cdot)$ der erste Erwartungswert und $H(\cdot)$ der negierte zweite Erwartungswert ist.
		\item Den Ausdruck $H(\btheta; \btheta^t)$ nennt man \keyword{Cross-Entropy} zwischen $\btheta$ und $\btheta^t$.
	\end{itemize}
\end{frame}


% Formulierung des EM-Algorithmus (Forts.)
\begin{frame}{Formulierung des EM-Algorithmus (Forts.)}{}
	\begin{itemize}
		\item Aufgrund der \keyword{Ungleichung von Gibbs} gilt
		\begin{equation}
			\label{eq:gibbs-ugl}
			H(\btheta; \btheta^t) \ge H(\btheta^t; \btheta^t).
		\end{equation}
		\item Daraus folgt
		\begin{align*}
			\ln p(\bX; \btheta) - \ln p(\bX; \btheta^t)
				&= Q(\btheta; \btheta^t) - Q(\btheta^t; \btheta^t) + H(\btheta; \btheta^t) - H(\btheta^t; \btheta^t)
		\\[2mm]
				&\overset{(\ref{eq:gibbs-ugl})}{\ge} Q(\btheta; \btheta^t) - Q(\btheta^t; \btheta^t) + H(\btheta^t; \btheta^t) - H(\btheta^t; \btheta^t)
		\\[2mm]
				&= Q(\btheta; \btheta^t) - Q(\btheta^t; \btheta^t).
		\end{align*}
		\item Diese Ungleichung zeigt: Wenn wir $Q(\btheta, \btheta^t)$ um einen Betrag $\delta$ verbessern, dann verbessern wir damit auch
			$\ln p(\bX; \btheta)$ \textit{um mindestens denselben Betrag}.
	\end{itemize}
\end{frame}


% Der EM-Algorithmus
\begin{frame}{Der EM-Algorithmus}{}
	Wir erhalten damit den folgenden iterativen Algorithmus:
	\begin{enumerate}
		\item Setze $t \coloneqq 0$ und initialisiere den Parametervektor $\btheta^t$.
		\item \textbf{E-Schritt (Expectation):}
		\begin{equation*}
			Q(\btheta; \btheta^t) \coloneqq \E_{\bZ \sim p(\bZ \mid \bX; \btheta^t)}\left[ \ln p(\bX, \bZ; \btheta) \right]
		\end{equation*}
		\item \textbf{M-Schritt (Maximization):}
		\begin{equation*}
			\btheta^{t+1} \coloneqq \argmax_{\btheta} Q(\btheta; \btheta^t)
		\end{equation*}
		\item Setze $t \coloneqq t + 1$
		\item Wenn der Algorithmus noch nicht konvergiert ist, gehe zu Schritt 2.
	\end{enumerate}
\end{frame}

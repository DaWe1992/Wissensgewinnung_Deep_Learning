% SUBSECTION: Motivation: Cocktail Party Problem
\subsection{Motivation: Cocktail Party Problem}

% Cocktail Party Problem
\begin{frame}{Cocktail Party Problem}{}
	Zur Motivation betrachten wir das folgende Problem:
	
	\begin{itemize}
		\item Auf einer Party sprechen $N$ Personen gleichzeitig.
		\item Im Raum werden Mikrofone aufgestellt, welche \textit{überlappende Kombinationen}
			der Stimmen der Sprecher aufzeichnen.
		\item Der Einfachheit halber nehmen wir an, dass sich ebenfalls $N$ Mikrofone im Raum befinden.
		\item Da die Entfernung der Sprecher zu den einzelnen Mikrofonen verschieden ist, handelt es sich hierbei um leicht
			\textit{verschiedene Kombinationen}.
	\end{itemize}
	
	\begin{boxGray}
		\begin{center}
			\textbf{Frage:} Können wir die Stimmen der $N$ Sprecher aus den $N$ Aufzeichnungen isolieren?
		\end{center}
	\end{boxGray}
\end{frame}


% Beispiel: Cocktail Party Problem
\begin{frame}{Beispiel: Cocktail Party Problem}{}
	tbd
\end{frame}


% SUBSECTION: Formalisierung des Problems
\subsection{Formalisierung des Problems}

% Mathematische Formulierung des Problems
\begin{frame}{Mathematische Formulierung des Problems}{}
	\begin{itemize}
		\item Es seien die Signale $\bs_t \in \R^N$ gegeben, welche von $N$ \textit{unabhängigen} \textbf{(!)}
			Quellen zum Zeitpunkt $t = 1, 2, \ldots, T$ erzeugt werden (das sind die Sprecher im Cocktail Party Problem).
		\item Allerdings beobachten wir nur die gemischten Daten $\bx_t \in \R^N$ zum Zeitpunkt $t$
			(das sind die Aufnahmen der Mikrofone).
		\item Das ICA-Modell nimmt an, dass es sich hierbei um \textit{Linearkombinationen} der Quellsignale handelt.
		\item Wir betrachten also das Modell
		\begin{equation}
			\bx_t = \bA \bs_t,
		\end{equation}
		wobei $\bA \in \R^{N \times N}$ die \keyword{Mixing Matrix} ist. Diese besteht aus den Koeffizienten der
		Linearkombinationen.
	\end{itemize}
\end{frame}


% Mathematische Formulierung des Problems (Forts.)
\begin{frame}{Mathematische Formulierung des Problems (Forts.)}{}
	\begin{itemize}
		\item Unser Ziel ist es, die Quellsignale $\bs$ aus den beobachteten Mischsignalen $\bx$ zu rekonstruieren.
		\item Wäre $\bA$ bekannt, so könnten wir in den meisten Fällen (falls die Inverse von $\bA$ existiert)
			$$\bs = \bA^{-1} \bx$$ berechnen und wären fertig.
		\item In der Regel ist $\bA$ unbekannt, wodurch das Problem deutlich schwerer wird.
		\item Wir definieren die \keyword{Unmixing Matrix} $\bW \coloneqq \bA^{-1}$.
		\item Mithilfe des ICA-Verfahrens werden wir diese Matrix approximieren.
			Mit ihr können wir dann die Quellsignale bestimmen, indem wir $\bs = \bW \bx$ berechnen.
	\end{itemize}
\end{frame}


% SUBSECTION: Annahmen und Limitationen der ICA
\subsection{Annahmen und Limitationen der ICA}

% Annahmen des Verfahrens
\begin{frame}{Annahmen des Verfahrens}{}
	Die folgenden Annahmen sind wesentlich für das ICA-Verfahren:

	\begin{enumerate}
		\item \textbf{Zentrale Annahme:} Die Quellsignale (ICs: Independent Components) sind
			\textit{stochastisch unabhängig}.
		\item Die ICs sind \textit{nicht normalverteilt} (dazu später mehr).
		\item Die Mixing Matrix $\bA$ ist quadratisch. Es gibt allerdings Verallgemeinerungen der ICA,
			die auch rechteckige Matrizen erlauben. Wir machen diese Annahme hier,
			da das Problem selbst in diesem vereinfachten Fall schon schwer genug ist.
	\end{enumerate}
	
	Im Cocktail Party Problem bedeutet Annahme 3, dass es genauso viele Mikrophone wie Personen gibt.
\end{frame}


% Limitationen des Verfahrens
\begin{frame}{Limitationen des Verfahrens}{}
	Es gibt eine Reihe von Einschränkungen (\textit{unidentifiabilities}) bei der Rekonstruktion der Quellsignale:

	\begin{enumerate}
		\item Die ursprüngliche \textit{Reihenfolge} der Signale kann nicht rekonstruiert werden.
			Sei dazu $\bP$ eine \textit{Permutationsmatrix}. Dann ist
			$\bx = \bA \bs = \bA \bP^{-1} \bP \bs$, wobei $\bP \bs$ die permutierten Quellsignale sind.
			$\bA \bP^{-1}$ ist lediglich eine neue zu schätzende Mixing Matrix.
			Wir können folglich kein Signal \textit{das erste} Quellsignal nennen.
		\item Die \textit{Skalierung} der Quellsignale kann nicht wiederhergestellt werden
			(im Cocktail Party Problem entspricht dies der Lautstärke/Amplitude der Signale), wie man sich leicht überlegt: \\
			Sei dazu $\alpha \in \R$, dann gilt $$\bx = \bA \bs = \alpha \bA \cdot \frac{1}{\alpha} \bs.$$
		\item Normalverteilte Signale können nicht aufgetrennt werden.
	\end{enumerate}
\end{frame}


% Warum sind Normalverteilungen verboten?
\begin{frame}{Warum sind Normalverteilungen verboten?}{}
	\begin{itemize}
		\item Wir betrachten ein Beispiel mit $N = 2$ und $\bs \sim N_{\0, \bI}$.
		\item Die Konturlinien der Standardnormalverteilung sind perfekte Kreise, die auf dem Koordinatenursprung zentriert sind.
		\item Insbesondere ist die Dichte \keyword{rotationsinvariant}.
		\item Für $\bx = \bA \bs$ gilt damit: $\bx \sim N_{\0, \bA \bA^\top}$. \\ \vspace*{2mm}
		\begin{myproof}
			Mit der Linearität des Erwartungswerts gilt $\E_{\bs}[\bx] = \E_{\bs}[\bA \bs] = \bA \E_{\bs}[s] = \0$.
			Die Streuungsmatrix ist
			\begin{align*}
				\D[\bx]
					&= \E_{\bs}\bigl[ \bx \bx^\top \bigr]
						= \E_{\bs}\bigl[ \bA \bs \bs^\top \bA^\top \bigr]
						= \bA \E\bigl[ \bs \bs^\top \bigr] \bA^\top
						= \bA \cdot \D[\bs] \cdot \bA^\top
			\\
					&= \bA \bA^\top.
			\end{align*}
			Also gilt die Behauptung.
		\end{myproof} 
	\end{itemize}
\end{frame}


% Warum sind Normalverteilungen verboten? (Forts.)
\begin{frame}{Warum sind Normalverteilungen verboten? (Forts.)}{}
	\begin{itemize}
		\item Sei nun $\bR \in \R^{N \times N}$ eine beliebige \textit{orthogonale Matrix}. \\ \vspace*{2mm}
		\textbf{Eigenschaften orthogonaler Matrizen:}
		\begin{itemize}
			\item Orthogonale Matrizen beschreiben eine Drehung/Spiegelung im Raum.
			\item Es gilt stets $\bR \bR^\top = \bR^\top \bR = \bI.$ Es ist also $\bR^{-1} = \bR^\top$.
			\item Es ist $\det \bR = 1$ für alle orthogonalen Matrizen $\bR$.
			\item Da $\bR$ nur eine Drehung beschreibt, gilt für alle $\bx \in \R^N$ die Identität
				$\Vert \bR \bx \Vert = \Vert \bx \Vert$.
		\end{itemize}
		\item Sei ferner $\bAtilde \coloneqq \bA \bR$ die Matrix, welche mit der von $\bR$ beschriebenen Drehung
			aus $\bA$ hervorgeht.
		\item Dann hat der Vektor $\bxtilde = \bAtilde \bs$ dieselbe Verteilung wie $\bx$. \\ \vspace*{2mm}
		\begin{myproof}
			Den Erwartungswert rechnet man analog zu oben aus. Aufgrund der Orthogonalität von $\bR$ gilt
			\begin{equation*}
				\E_{\bs}\bigl[ \bxtilde \bxtilde^\top \bigr]
					= \E_{\bs}\bigl[ \bAtilde \bs \bs^\top \bAtilde^\top \bigr]
					= \E_{\bs}\bigl[ \bA \bR \bs \bs^\top \bR^\top \bA^\top \bigr]
					= \bA \bR \cdot \D[\bs] \cdot \bR^\top \bA^\top
					= \bA \bR \bR^\top \bA
					= \bA \bA^\top,
			\end{equation*}
			sodass $\bxtilde \sim N_{\0, \bA \bA^\top}$ wie behauptet gilt.
		\end{myproof}
	\end{itemize}
\end{frame}


% Warum sind Normalverteilungen verboten? (Forts.)
\begin{frame}{Warum sind Normalverteilungen verboten? (Forts.)}{}
	\begin{itemize}
		\item Wie die voranstehenden Ausführungen zeigen, können wir in diesem Fall nicht rekonstruieren,
			ob die Daten mit der Matrix $\bA$ oder $\bAtilde$ gemixt wurden.
		\item Für normalverteilte Daten können wir folglich die Komponenten \textit{nur bis auf eine Drehung rekonstruieren}.
	\end{itemize}
\end{frame}


% Ein einfaches Beispiel
\begin{frame}{Ein einfaches Beispiel}{}
	Wir betrachten ein einfaches zweidimensionales Beispiel für den Fall, dass die Quellsignale $s_1$ und $s_2$
	gleichverteilt sind:
	\begin{equation*}
		p(s_i) \coloneqq \begin{cases}
			\frac{1}{2 \sqrt{3}} 	& \text{wenn}\ \vert s_i \vert \le \sqrt{3} \\
			0 					& \text{sonst}
		\end{cases}
	\end{equation*}

	\textbf{Bemerkungen:}
	\begin{itemize}
		\item Der Erwartungswert der Zufallsvariablen $s_i$ $(i = 1, 2)$ ist jeweils 0. \\ \vspace*{2mm}
		\begin{myproof}
			Für die Gleichverteilung gilt $\E[s_i] = \frac{-\sqrt{3} + \sqrt{3}}{2} = 0$ (siehe Tabelle im Kapitel `Statistik').
		\end{myproof} \\ \vspace*{1mm}
		\item Die Varianzen sind jeweils 1. \\ \vspace*{2mm}
		\begin{myproof}
			Es ist $\V[s_i]
				= \frac{\bigl( \sqrt{3} + \sqrt{3} \bigr)^2}{12}
				= \frac{\bigl( 2 \sqrt{3} \bigr)^2}{12} = \frac{12}{12} = 1$
			(siehe Tabelle im Kapitel `Statistik').
		\end{myproof}
	\end{itemize}
\end{frame}

% SUBSECTION: Whitening
\subsection{Whitening}

% Whitening
\begin{frame}{Whitening}{}
	Whitening wird häufig vor dem eigentlichen ICA-Algorithmus angewendet:
	\begin{mydef}[def:whitening]
		\keyword{Whitening} bezeichnet eine \textit{lineare Transformation} $\bV$,
		bei der zentrierte Zufallsvariablen $\bx \in \R^N$
		(zentriert bedeutet $\E[\bx] = \0$) so zu neuen Zufallsvariablen $\bz \coloneqq \bV \bx \in \R^N$ transformiert werden,
		dass $\D[\bz] = \E\bigl[ \bz \bz^\top \bigr] = \bI$ gilt.
	\end{mydef}

	In Worten bedeutet dies, dass die Komponenten von $\bz$ untereinander unkorreliert und die Varianzen jeweils Eins sind,
	also $$\E[z_i z_j] = \delta_{ij},$$ wobei $\delta_{ij} \coloneqq
	\begin{cases} 1 & \text{falls}\ i = j \\ 0 & \text{falls}\ i \ne j \end{cases}$ das \keyword{Kronecker-Delta} bezeichnet.
\end{frame}


% Whitening (Forts.)
\begin{frame}{Whitening (Forts.)}{}
	\begin{itemize}
		\item Eine solche lineare Transformation ist z.\,B. gegeben durch $\bV \coloneqq \bD^{-\frac{1}{2}} \bE^\top$,
			wobei $\bE \in \R^{N \times N}$ eine orthogonale Matrix ist,
			deren Spalten aus \textit{normalisierten Eigenvektoren} von $\D[\bx]$ bestehen und
			$\bD \coloneqq \diag\bigl( d_1, d_2, \ldots, d_N \bigr) \in \R^{N \times N}$ eine Diagonalmatrix mit den
			zugehörigen \textit{Eigenwerten} ist. \\ \vspace*{2mm}
		\begin{myproof}
			Wir rechnen nach, dass die Kovarianzmatrix von $\bz = \bV \bx$ tatsächlich die Einheitsmatrix ist.
			Da $\E\bigl[ \bx \bx^\top \bigr]$ in praktischen Anwendungsfällen in der Regel positiv definit ist,
			gilt $\E\bigl[ \bx \bx^\top \bigr] = \bE \bD \bE^\top$. Damit gilt tatsächlich
			\begin{equation*}
				\E\bigl[ \bz \bz^\top \bigr]
					= \E\bigl[ \bV \bx \bx^\top \bV^\top \bigr]
					= \bV \E\bigl[ \bx \bx^\top \bigr] \bV^\top
					= \bD^{-\frac{1}{2}} \bE^\top \bE \bD \bE^\top \bE \bD^{-\frac{1}{2}}
					= \bI.
			\end{equation*}
		\end{myproof}
		\item Aus der Zentriertheit von $\bx$ folgt, dass auch $\bz$ zentriert ist. \\ \vspace*{2mm}
		\begin{myproof}
			Wieder mit der Linearität des Erwartungswerts gilt $\E[\bz] = \E[\bV \bx] = \bV \E[\bx] = \0$.
		\end{myproof}
	\end{itemize}
\end{frame}

% Whitening löst bereits das halbe Problem
\begin{frame}{Whitening löst bereits das halbe Problem}{}
	\begin{itemize}
		\item Wir nehmen im Folgenden an, dass bereits ein Whitening stattgefunden hat.
		\item Whitening transformiert die Mixing Matrix $\bA$ in eine neue Matrix $\bAtilde$, denn
			$$\bz = \bV \bx = \bV \bA \bs = \bAtilde \bs.$$
		\item Man könnte nun hoffen, dass Whitening das Problem bereits löst,
			da Unkorreliertheit und Unabhängigkeit verwandt sind. \textbf{Dem ist jedoch nicht so!}
		\item Sei dazu $\bU$ eine orthogonale Matrix und $\by \coloneqq \bU \bz$.
		\item Aufgrund der Orthogonalität von $\bU$ gilt
			$$\E\bigl[ \by \by^\top \bigr] = \E\bigl[ \bU \by \by^\top \bU^\top \bigr] = \bU \bI \bU^\top = \bI.$$
			Somit ist auch $\by$ white. Die Komponenten sind durch Whitening nur bis auf eine Drehung eindeutig!
	\end{itemize}
\end{frame}


% Whitening als Vorverarbeitungsschritt
\begin{frame}{Whitening als Vorverarbeitungsschritt}{}
	\begin{itemize}
		\item Die Whitening Operation eignet sich jedoch sehr gut als Vorverarbeitungsschritt,
			denn durch sie wird die Mixing Matrix $\bAtilde = \bV \bA$ orthogonal. \\ \vspace*{2mm}
		\begin{myproof}
			Da die Daten $\bz$ white sind, gilt $\E\bigl[ \bz \bz^\top \bigr] = \bI$. Es gilt aber auch
			\begin{equation*}
				\E\bigl[ \bz \bz^\top \bigr]
					= \E\bigl[ \bAtilde \bs \bs^\top \bAtilde^\top \bigr]
					= \bAtilde \bAtilde^\top.
			\end{equation*}
			Also gilt $\bAtilde \bAtilde^\top = \bI$, woraus die Orthogonalität von $\bAtilde$ folgt.
		\end{myproof} \\ \vspace*{1mm}
		\item Das heißt wir können uns bei der Suche nach der Mixing Matrix \textbf{auf orthogonale Matrizen beschränken!}
		\item Eine orthogonale $N \times N$ Matrix hat nur $N (N - 1) / 2$ Freiheitsgrade
			(im Gegensatz zu $N^2$ Freiheitsgraden für die ursprüngliche Matrix),
			z.\,B. wird eine zweidimensionale orthogonale Matrix eindeutig durch den Winkel der Drehung beschrieben.
	\end{itemize}
\end{frame}


% Warum Normalverteilungen verboten sind!
\begin{frame}{Warum Normalverteilungen verboten sind!}{}
	\begin{itemize}
		\item Durch Whitening können wir eine weitere Begründung dafür angeben, warum Normalverteilungen verboten sind.
		\item Wir nehmen an, dass die gemeinsame Verteilung der Quellsignale $s_1$ und $s_2$
			durch eine Normalverteilung beschrieben werden:
		\begin{equation*}
			p(s_1, s_2)
				= \frac{1}{2 \pi} \exp\left( -\frac{s_1^2 + s_2^2}{2} \right)
				= \frac{1}{2 \pi} \exp\left( -\frac{\Vert \bs \Vert^2}{2} \right)
		\end{equation*}
		\item Die Mixing Matrix $\bA$ kann als orthogonal angenommen werden, da die Daten $\bx$ als white vorausgesetzt sind.
	\end{itemize}
\end{frame}


% Warum Normalverteilungen verboten sind! (Forts.)
\begin{frame}{Warum Normalverteilungen verboten sind! (Forts.)}{}
	Mit dem Transformationssatz \ref{th:transformationssatz} für Dichten erhalten wir daraus
	die Verteilung von $x_1$ und $x_2$:
	\begin{align*}
		p(x_1, x_2)
			&= \frac{1}{2 \pi} \exp\left( -\frac{\Vert \bA^{-1} \bx \Vert^2}{2} \right) \vert \det \bA^{-1} \vert
	\\[4mm]
			&= \frac{1}{2 \pi} \exp\left( -\frac{\Vert \bA^\top \bx \Vert^2}{2} \right) \vert \det \bA^\top \vert
			&& \rem{Aufgrund der Orthogonalität gilt $\bA^{-1} = \bA^\top$.}
	\\[4mm]
			&= \frac{1}{2 \pi} \exp\left( -\frac{\Vert \bx \Vert^2}{2} \right)
			&& \rem{Aufgrund der Orthogonalität gilt $\Vert \bA^\top \bx \Vert = \Vert \bx \Vert$ und $\det \bA^\top = 1$.}
	\end{align*}
\end{frame}


% Warum Normalverteilungen verboten sind! (Forts.)
\begin{frame}{Warum Normalverteilungen verboten sind! (Forts.)}{}
	\begin{itemize}
		\item Wir sehen also, dass die Dichte der Quellsignale und der gemischten Signale identisch ist.
		\item Damit haben wir keine Chance, die Mixing Matrix aus den Daten zu schätzen.
	\end{itemize}
\end{frame}

% SUBSECTION: ICA durch Maximum Likelihood Schätzung
\subsection{ICA durch Maximum Likelihood Schätzung}

% ICA Lösungsverfahren
\begin{frame}{ICA Lösungsverfahren}{}
	Es gibt verschiedene Ansätze mit denen man die unabhängigen Komponenten bestimmen kann:
	\begin{enumerate}
		\item Maximierung der Likelihood Funktion.
		\item Maximierung der \textit{Non-Gaussianity}: Finde die unabhängigen Komponenten so,
			dass deren Verteilung maximal nicht-normal ist.
		\begin{itemize}
			\item Basiert auf dem \keyword{zentralen Grenzwertsatz}: Eine Summe unabhängiger Zufallsvariablen ist
				(unter gewissen Voraussetzungen) zunehmend normalverteilt.
			\item Hierzu muss man eine Maßzahl finden, welche die Nicht-Normalität einer Verteilung misst.
			\item Beispiele: \keyword{Kurtosis}, \keyword{Negentropy}
		\end{itemize}
		\item Minimierung der Mutual Information.
	\end{enumerate}
\end{frame}


% ICA Likelihood Funktion
\begin{frame}{ICA Likelihood Funktion}{}
	Wir leiten im Folgenden die Likelihood Funktion für die ICA her:
	
	\begin{itemize}
		\item Da die Komponenten $s_n$ \textit{stochastisch unabhängig} sind, finden wir deren gemeinsame Dichte als
			\textit{Produktdichte} der Komponenten $$p_{\bs}(\bs) = \prod_{n=1}^N p_n(s_n).$$
		\item Wir müssen dies allerdings mittels der beobachteten Daten $\bx = \bA \bs$ ausdrücken, da $\bs$ \textit{latent} ist.
			Hierzu benutzen wir den Transformationssatz \ref{th:transformationssatz} und erhalten mit
			$\bs = \bA^{-1} \bx = \bW \bx$ und
			$\bW = \bigl( \bw_1, \bw_2, \ldots, \bw_N \bigr)^\top$ die Dichte von $\bx$ als
		\begin{equation} \label{eq:product-dist-observed-data}
			p_{\bx}(\bx) = \vert \det \bW \vert \cdot \prod_{n=1}^N p_n\bigl( (\bw_n)^\top \bx \bigr).
		\end{equation}
	\end{itemize}
\end{frame}


% ICA Likelihood Funktion (Forts.)
\begin{frame}{ICA Likelihood Funktion (Forts.)}{}	
	\begin{itemize}
		\item Wir nehmen nun an, dass wir $T$ Beobachtungen $\bx_1, \bx_2, \ldots, \bx_T$ von $\bx$ haben.
		\item Die Likelihood Funktion ergibt sich als das Produkt der Dichte in \eqref{eq:product-dist-observed-data}
		über die $T$ Beobachtungen
		\begin{equation*}
			L(\bW)
				= \prod_{t=1}^T \left( \vert \det \bW \vert \cdot \prod_{n=1}^N p_n\bigl( (\bw_n)^\top \bx_t \bigr) \right)
				= \vert \det \bW \vert^T \cdot \prod_{t=1}^T \prod_{n=1}^N p_n\bigl( (\bw_n)^\top \bx_t \bigr).
		\end{equation*}
		\item Im Folgenden betrachten wir die Log-Likelihood Funktion
		\begin{align*}
			l(\bW)
				&= \ln\left(
					\vert \det \bW \vert^T \cdot \prod_{t=1}^T \prod_{n=1}^N p_n\bigl( (\bw_n)^\top \bx_t \bigr)
				\right)
					= \ln \vert \det \bW \vert^T + \ln\left(
						\prod_{t=1}^T \prod_{n=1}^N p_n\bigl( (\bw_n)^\top \bx_t \bigr)
					\right)
		\\[2mm]
				&= T \cdot \ln \vert \det \bW \vert + \sum_{t=1}^T \sum_{n=1}^N \ln p_n\bigl( (\bw_n)^\top \bx_t \bigr).
		\end{align*}
	\end{itemize}
\end{frame}


% Festlegung der Dichte der Quellsignale
\begin{frame}{Festlegung der Dichte der Quellsignale}{}
	\begin{itemize}
		\item Um die Likelihood Funktion optimieren zu können, müssen wir noch die Dichten $p_n(s_n)$ ($n = 1, 2, \ldots, N$)
			für die Quelldaten festlegen.
		\item Bei der Wahl dürfen wir aufgrund obiger Ausführungen \textit{keine Normalverteilungen} verwenden.
		\item Aufgrund der Ähnlichkeit zur Normalverteilung schlägt \cite{Ng2023} die \keyword{logistische Verteilung} vor.
		\item Die Verteilungsfunktion (CDF) ist folgendermaßen definiert:
		\begin{equation*}
			F(s_0) = p(s \le s_0) = \frac{1}{1 + e^{-s_0}} \eqqcolon \sigma(s)
		\end{equation*}
		\item Damit ergibt sich für die Dichte $p_n(s_n) = \sigma'(s) = \sigma(s) \cdot \bigl( 1 - \sigma(s) \bigr)$
			und für die Log-Likelihood
		\begin{equation*}
			T \cdot \ln \vert \det \bW \vert + \sum_{t=1}^T \sum_{n=1}^N \ln \sigma'\bigl( (\bw_n)^\top \bx_t \bigr).
		\end{equation*}
	\end{itemize}
\end{frame}


% Bestimmung des Gradienten
\begin{frame}{Bestimmung des Gradienten}{}
	\begin{mytheo}[th:gradient-det]
		Es sei $\bW \in \R^{N \times N}$. Dann ist
		$\frac{\partial}{\partial \bW} \det \bW = \bigl( \bW^\top \bigr)^{-1} \det \bW$.
	\end{mytheo}

	\begin{myproof}
		Mit dem Adjunktensatz der linearen Algebra gilt $\bW^{-1} = \frac{1}{\det \bW} \adj(\bW)$.
		Hierbei bezeichnet $\adj(\bW) = \bigl( W_{ij} \bigr)_{i,j = 1, 2, \ldots, N}$ die zu $\bW$ \textbf{adjungierte Matrix}.
		Die \textbf{Kofaktoren} $W_{ij}$ erhält man, indem man die Determinante der
		$(N - 1) \times (N - 1)$ Untermatrix von $\bW$ berechnet, die aus der Streichung der $i$-ten Zeile und $j$-ten Spalte
		von $\bW$ hervorgeht. Dieses Ergebnis muss schließlich noch mit $(-1)^{i + j}$ multipliziert werden.
		
		Die Determinante kann nun in Form der Kofaktoren ausgedrückt werden
		(Stichwort: \textbf{Entwicklungssatz von Laplace}). Wir erhalten
		\begin{equation*}
			\det \bW = \sum_{k = 1}^N w_{ik} W_{ik}
		\end{equation*}
		Zeile $i$ kann dabei beliebig gewählt werden. Entscheidend ist nun, dass die Kofaktoren per Definition
		nicht von den Einträgen der Zeile $i$ der Matrix abhängen, sodass die Determinante eine lineare Funktion
		dieser Einträge ist. Wir erhalten damit
		$\frac{\partial \det \bW}{\partial w_{ij}} = W_{ij}$ und schließlich
		$\frac{\partial \det \bW}{\partial \bW} = \adj(\bW)^\top = \bigl( \bW^\top \bigr)^{-1} \det \bW$.
		Die letzte Gleichung gilt dabei wegen des Adjunktensatzes.
	\end{myproof}
\end{frame}


% Bestimmung des Gradienten (Forts.)
\begin{frame}{Bestimmung des Gradienten (Forts.)}{}
	Mit diesem Ergebnis können wir nun eine einfache Regel für den stochastischen Gradientenaufstieg angeben:
	\begin{itemize}
		\item Sei dazu ein Trainingsbeispiel $\bx \in \R^N$  und die Lernrate $\alpha \in \R$ gegeben.
		\item Die Update-Regel lautet dann
		\begin{equation}
			\bW^{\text{neu}} \coloneqq \bW + \alpha \left[
				\begin{pmatrix}
					1 - 2 \sigma\bigl( (\bw_1)^\top \bx \bigr) \\
					1 - 2 \sigma\bigl( (\bw_2)^\top \bx \bigr) \\
					\vdots \\
					1 - 2 \sigma\bigl( (\bw_N)^\top \bx \bigr)
				\end{pmatrix} \bx^\top + \bigl( \bW^\top \bigr)^{-1}
			\right].
		\end{equation}
	\end{itemize}
\end{frame}

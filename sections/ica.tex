\section{Independent Component Analysis}

\subsection{Motivation: Cocktail Party Problem}

% Cocktail Party Problem
\begin{frame}{Cocktail Party Problem}{}
	Zur Motivation betrachten wir das folgende Problem:
	
	\begin{itemize}
		\item Auf einer Party sprechen $N$ Personen gleichzeitig.
		\item Im Raum werden Mikrofone aufgestellt, welche \textit{û¥berlappende Kombinationen} der Stimmen der Sprecher aufzeichnen.
		\item Der Einfachheit halber nehmen wir an, dass sich ebenfalls $N$ Mikrofone im Raum befinden.
		\item Da die Entfernung der Sprecher zu den einzelnen Mikrofonen verschieden ist, handelt es sich hierbei um leicht
			\textit{verschiedene Kombinationen}.
	\end{itemize}

	\textbf{KûÑnnen wir die Stimmen der $N$ Sprecher aus den $N$ Aufzeichnungen isolieren?}
\end{frame}

\subsection{Formalisierung des Problems}

% Mathematische Formulierung des Problems
\begin{frame}{Mathematische Formulierung des Problems}{}
	\begin{itemize}
		\item Es seien die Signale $\bs(t) \in \R^N$ gegeben, welche von $N$ \textit{unabhûÊngigen} (!) Quellen zum Zeitpunkt $t = 1, 2, \ldots, T$
			erzeugt werden (die Sprecher im Cocktail Party Problem).
		\item Allerdings beobachten wir nur die gemischten Daten $\bx(t)$ zum Zeitpunkt $t$ (die Aufnahmen der Mikrofone).
		\item Das ICA-Modell nimmt an, dass es sich hierbei um \textit{Linearkombinationen} der Quellsignale handelt.
		\item Wir betrachten also das Modell
		\begin{equation}
			\bx(t) = \bA \bs(t),
		\end{equation}
		wobei $\bA$ die \keyword{Mixing Matrix} ist.
	\end{itemize}
\end{frame}


% Mathematische Formulierung des Problems (Forts.)
\begin{frame}{Mathematische Formulierung des Problems (Forts.)}{}
	\begin{itemize}
		\item Unser Ziel ist es, die Quellsignale $\bs$ aus den beobachteten Mischsignalen $\bx$ zu rekonstruieren.
		\item WûÊre $\bA$ bekannt, so kûÑnnten wir in den meisten FûÊllen $$\bs = \bA^{-1} \bx$$ berechnen und wûÊren fertig.
		\item In der Regel ist $\bA$ unbekannt, wodurch das Problem deutlich schwerer wird.
		\item Wir definieren die \keyword{Unmixing Matrix} $\bB \coloneqq \bA^{-1}$.
		\item Mithilfe des ICA-Verfahrens werden wir diese Matrix approximieren. Mit ihr kûÑnnen wir dann die Quellsignale bestimmen, indem wir
			$\bs = \bB \bx$ berechnen.
	\end{itemize}
\end{frame}


\subsection{Annahmen und Limitationen von ICA}

% Annahmen des Verfahrens
\begin{frame}{Annahmen des Verfahrens}{}
	Die folgenden Annahmen sind wesentlich fû¥r das ICA-Verfahren:

	\begin{enumerate}
		\item \textbf{Zentrale Annahme:} Die Quellsignale (ICs: Independent Components) sind \textit{stochastisch unabhûÊngig}.
		\item Die ICs sind \textit{nicht normalverteilt} (dazu spûÊter mehr).
		\item Die Mixing Matrix $\bA$ ist quadratisch. Es gibt allerdings Verallgemeinerungen der ICA, die auch rechteckige Matrizen erlauben.
			Wir machen diese Annahme hier, da das Problem selbst in diesem vereinfachten Fall schon schwer genug ist.
	\end{enumerate}
\end{frame}


% Limitationen des Verfahrens
\begin{frame}{Limitationen des Verfahrens}{}
	Es gibt eine Reihe von EinschrûÊnkungen bei der Rekonstruktion der Quellsignale:

	\begin{enumerate}
		\item Die ursprû¥ngliche \textit{Reihenfolge} der Signale kann nicht rekonstruiert werden.
			Sei dazu $\bP$ eine \textit{Permutationsmatrix}. Dann ist
			$\bx = \bA \bs = \bA \bP^{-1} \bP \bs$, wobei $\bP \bs$ die permutierten Quellsignale sind.
			$\bA \bP^{-1}$ ist lediglich eine neue zu schûÊtzende Mixing Matrix. Wir kûÑnnen folglich kein Signal \textit{das erste} Quellsignal nennen.
		\item Die \textit{Skalierung} der Quellsignale kann nicht wiederhergestellt werden
			(im Cocktail Party Problem entspricht dies der LautstûÊrke/Amplitude der Signale), wie man sich leicht û¥berlegt: \\
			Sei dazu $\alpha \in \R$, dann gilt $$\bx = \bA \bs = \alpha \bA \cdot \frac{1}{\alpha} \bs.$$
		\item Normalverteilte Signale kûÑnnen nicht aufgetrennt werden.
	\end{enumerate}
\end{frame}


% Warum sind Normalverteilungen verboten?
\begin{frame}{Warum sind Normalverteilungen verboten?}{}
	\begin{itemize}
		\item Wir betrachten ein Beispiel mit $N = 2$ und $\bs \sim N_{\0, \bI}$.
		\item Die Konturlinien der Standardnormalverteilung sind perfekte Kreise, die auf dem Koordinatenursprung zentriert sind.
		\item Insbesondere ist die Dichte \keyword{rotationsinvariant}.
		\item Fû¥r $\bx = \bA \bs$ gilt damit: $\bx \sim N_{\0, \bA \bA^\top}$. \\ \vspace*{2mm}
		\begin{myproof}
			Mit der LinearitûÊt des Erwartungswerts gilt $\E_{\bs}[\bx] = \E_{\bs}[\bA \bs] = \bA \E_{\bs}[s] = \0$.
			Die Streuungsmatrix ist
			\begin{align*}
				\D[\bx]
					&= \E_{\bs}\bigl[ \bx \bx^\top \bigr]
						= \E_{\bs}\bigl[ \bA \bs \bs^\top \bA^\top \bigr]
						= \bA \E\bigl[ \bs \bs^\top \bigr] \bA^\top
						= \bA \cdot \D[\bs] \cdot \bA^\top
			\\
					&= \bA \bA^\top.
			\end{align*}
			Also gilt die Behauptung.
		\end{myproof} 
	\end{itemize}
\end{frame}


% Warum sind Normalverteilungen verboten? (Forts.)
\begin{frame}{Warum sind Normalverteilungen verboten? (Forts.)}{}
	\begin{itemize}
		\item Sei nun $\bR \in \R^{N \times N}$ eine beliebige \textit{orthogonale Matrix}. \\ \vspace*{2mm}
		\textbf{Eigenschaften orthogonaler Matrizen:}
		\begin{itemize}
			\item Orthogonale Matrizen beschreiben eine Drehung/Spiegelung im Raum.
			\item Es gilt stets $\bR \bR^\top = \bR^\top \bR = \bI.$ Es ist also $\bR^{-1} = \bR^\top$.
			\item Es ist $\det \bR = 1$ fû¥r alle orthogonalen Matrizen $\bR$.
			\item Da $\bR$ nur eine Drehung beschreibt, gilt fû¥r alle $\bx \in \R^N$ die IdentitûÊt $\Vert \bR \bx \Vert = \Vert \bx \Vert$.
		\end{itemize}
		\item Sei ferner $\bAtilde \coloneqq \bA \bR$ die Matrix, welche mit der von $\bR$ beschriebenen Drehung aus $\bA$ hervorgeht.
		\item Dann hat der Vektor $\bxtilde = \bAtilde \bs$ dieselbe Verteilung wie $\bx$. \\ \vspace*{2mm}
		\begin{myproof}
			Den Erwartungswert rechnet man analog zu oben aus. Aufgrund der OrthogonalitûÊt von $\bR$ gilt
			\begin{equation*}
				\E_{\bs}\bigl[ \bxtilde \bxtilde^\top \bigr]
					= \E_{\bs}\bigl[ \bAtilde \bs \bs^\top \bAtilde^\top \bigr]
					= \E_{\bs}\bigl[ \bA \bR \bs \bs^\top \bR^\top \bA^\top \bigr]
					= \bA \bR \cdot \D[\bs] \cdot \bR^\top \bA^\top
					= \bA \bR \bR^\top \bA
					= \bA \bA^\top,
			\end{equation*}
			sodass $\bxtilde \sim N_{\0, \bA \bA^\top}$ wie behauptet gilt.
		\end{myproof}
	\end{itemize}
\end{frame}


% Warum sind Normalverteilungen verboten? (Forts.)
\begin{frame}{Warum sind Normalverteilungen verboten? (Forts.)}{}
	\begin{itemize}
		\item Die voranstehenden Ausfû¥hrungen zeigen, kûÑnnen wir in diesem Fall nicht rekonstruieren, ob die Daten mit der Matrix
			$\bA$ oder $\bAtilde$ gemixt wurden.
		\item Fû¥r normalverteilte Daten kûÑnnen wir folglich die Komponenten \textit{nur bis auf eine Drehung rekonstruieren}.
	\end{itemize}
\end{frame}

\begin{frame}{Ein einfaches Beispiel}{}
	Wir betrachten ein einfaches zweidimensionales Beispiel fû¥r den Fall, dass die Quellsignale $s_1$ und $s_2$ gleichverteilt sind:
	\begin{equation*}
		p(s_i) \coloneqq \begin{cases}
			\frac{1}{2 \sqrt{3}} 	& \text{wenn}\ \vert s_i \vert \le \sqrt{3} \\
			0 				& \text{sonst}
		\end{cases}
	\end{equation*}

	\textbf{Bemerkungen:}
	\begin{itemize}
		\item Der Erwartungswert der Zufallsvariablen $s_i$ $(i = 1, 2)$ ist jeweils 0. \\ \vspace*{2mm}
		\begin{myproof}
			Fû¥r die Gleichverteilung gilt $\E[s_i] = \frac{-\sqrt{3} + \sqrt{3}}{2} = 0$ (siehe Tabelle im Kapitel `Statistik').
		\end{myproof} \\ \vspace*{1mm}
		\item Die Varianzen sind jeweils 1. \\ \vspace*{2mm}
		\begin{myproof}
			Es ist $\V[s_i] = \frac{\bigl( \sqrt{3} + \sqrt{3} \bigr)^2}{12} = \frac{\bigl( 2 \sqrt{3} \bigr)^2}{12} = \frac{12}{12} = 1$
			(siehe Tabelle im Kapitel `Statistik').
		\end{myproof}
	\end{itemize}
\end{frame}

\subsection{Whitening}

% Whitening
\begin{frame}{Whitening}{}
	Whitening wird hûÊufig vor dem eigentlichen ICA-Algorithmus angewendet:
	\begin{mydef}[def:whitening]
		\keyword{Whitening} bezeichnet eine \textit{lineare Transformation} $\bV$, bei der zentrierte Zufallsvariablen $\bx \in \R^N$
		(zentriert bedeutet $\E[\bx] = \0$) so zu neuen Zufallsvariablen $\bz \coloneqq \bV \bx \in \R^N$ transformiert werden, dass
		$\D[\bz] = \E\bigl[ \bz \bz^\top \bigr] = \bI$ gilt.
	\end{mydef}

	In Worten bedeutet dies, dass die Komponenten von $\bz$ untereinander unkorreliert und die Varianzen jeweils Eins sind,
	also $$\E[z_i z_j] = \delta_{ij},$$ wobei $\delta_{ij} \coloneqq \begin{cases} 1 & \text{falls}\ i = j \\ 0 & \text{falls}\ i \ne j \end{cases}$
	das \keyword{Kronecker-Delta} bezeichnet.
\end{frame}


% Whitening (Forts.)
\begin{frame}{Whitening (Forts.)}{}
	\begin{itemize}
		\item Eine solche lineare Transformation ist z.\,B. gegeben durch $\bV \coloneqq \bD^{-\frac{1}{2}} \bE^\top$, wobei $\bE \in \R^{N \times N}$
			eine orthogonale Matrix ist, deren Spalten aus \textit{normalisierten Eigenvektoren} von $\D[\bx]$ bestehen und
			$\bD \coloneqq \diag\bigl( d_1, d_2, \ldots, d_N \bigr) \in \R^{N \times N}$ eine Diagonalmatrix mit den zugehûÑrigen \textit{Eigenwerten} ist.
			\\ \vspace*{2mm}
		\begin{myproof}
			Wir rechnen nach, dass die Kovarianzmatrix von $\bz = \bV \bx$ tatsûÊchlich die Einheitsmatrix ist. Da $\E\bigl[ \bx \bx^\top \bigr]$
			in praktischen AnwendungsfûÊllen in der Regel positiv definit ist, gilt $\E\bigl[ \bx \bx^\top \bigr] = \bE \bD \bE^\top$. Damit gilt tatsûÊchlich
			\begin{equation*}
				\E\bigl[ \bz \bz^\top \bigr]
					= \E\bigl[ \bV \bx \bx^\top \bV^\top \bigr]
					= \bV \E\bigl[ \bx \bx^\top \bigr] \bV^\top
					= \bD^{-\frac{1}{2}} \bE^\top \bE \bD \bE^\top \bE \bD^{-\frac{1}{2}}
					= \bI.
			\end{equation*}
		\end{myproof}
		\item Aus der Zentriertheit von $\bx$ folgt, dass auch $\bz$ zentriert ist. \\ \vspace*{2mm}
		\begin{myproof}
			Wieder mit der LinearitûÊt des Erwartungswerts gilt $\E[\bz] = \E[\bV \bx] = \bV \E[\bx] = \0$.
		\end{myproof}
	\end{itemize}
\end{frame}

% Whitening lûÑst bereits das halbe Problem
\begin{frame}{Whitening lûÑst bereits das halbe Problem}{}
	\begin{itemize}
		\item Wir nehmen im Folgenden an, dass bereits ein Whitening stattgefunden hat.
		\item Whitening transformiert die Mixing Matrix $\bA$ in eine neue Matrix $\bAtilde$, denn
			$$\bz = \bV \bx = \bV \bA \bs = \bAtilde \bs.$$
		\item Man kûÑnnte nun hoffen, dass Whitening das Problem bereits lûÑst, da Unkorreliertheit und UnabhûÊngigkeit verwandt sind.
			\textbf{Dem ist jedoch nicht so!}
		\item Sei dazu $\bU$ eine orthogonale Matrix und $\by \coloneqq \bU \bz$.
		\item Aufgrund der OrthogonalitûÊt von $\bU$ gilt
			$$\E\bigl[ \by \by^\top \bigr] = \E\bigl[ \bU \by \by^\top \bU^\top \bigr] = \bU \bI \bU^\top = \bI.$$
			Somit ist auch $\by$ white. Die Komponenten sind durch Whitening nur bis auf eine Drehung eindeutig!
	\end{itemize}
\end{frame}


% Whitening als Vorverarbeitungsschritt
\begin{frame}{Whitening als Vorverarbeitungsschritt}{}
	\begin{itemize}
		\item Die Whitening Operation eignet sich jedoch sehr gut als Vorverarbeitungsschritt, denn durch sie wird die Mixing Matrix
			$\bAtilde = \bV \bA$ orthogonal. \\ \vspace*{2mm}
		\begin{myproof}
			Da die Daten $\bz$ white sind, gilt $\E\bigl[ \bz \bz^\top \bigr] = \bI$. Es gilt aber auch
			\begin{equation*}
				\E\bigl[ \bz \bz^\top \bigr]
					= \E\bigl[ \bAtilde \bs \bs^\top \bAtilde^\top \bigr]
					= \bAtilde \bAtilde^\top.
			\end{equation*}
			Also gilt $\bAtilde \bAtilde^\top = \bI$, woraus die OrthogonalitûÊt von $\bAtilde$ folgt.
		\end{myproof} \\ \vspace*{1mm}
		\item Das heiût wir kûÑnnen uns bei der Suche nach der Mixing Matrix \textbf{auf orthogonale Matrizen beschrûÊnken!}
		\item Eine orthogonale $N \times N$ Matrix hat nur $N (N - 1) / 2$ Freiheitsgrade
			(im Gegensatz zu $N^2$ Freiheitsgraden fû¥r die ursprû¥ngliche Matrix),
			z.\,B. wird eine zweidimensionale orthogonale Matrix eindeutig durch den Winkel der Drehung beschrieben.
	\end{itemize}
\end{frame}


% Warum Normalverteilungen verboten sind!
\begin{frame}{Warum Normalverteilungen verboten sind!}{}
	\begin{itemize}
		\item Durch Whitening kûÑnnen wir eine weitere Begrû¥ndung dafû¥r angeben, warum Normalverteilungen verboten sind.
		\item Wir nehmen an, dass die gemeinsame Verteilung der Quellsignale $s_1$ und $s_2$ durch eine Normalverteilung beschrieben werden:
		\begin{equation*}
			p(s_1, s_2)
				= \frac{1}{2 \pi} \exp\left( -\frac{s_1^2 + s_2^2}{2} \right)
				= \frac{1}{2 \pi} \exp\left( -\frac{\Vert \bs \Vert^2}{2} \right)
		\end{equation*}
		\item Die Mixing Matrix $\bA$ kann als orthogonal angenommen werden, da die Daten $\bx$ als white vorausgesetzt sind.
	\end{itemize}
\end{frame}


% Warum Normalverteilungen verboten sind! (Forts.)
\begin{frame}{Warum Normalverteilungen verboten sind! (Forts.)}{}
	Mit dem Transformationssatz \ref{th:transformationssatz} fû¥r Dichten erhalten wir daraus die Verteilung von $x_1$ und $x_2$:
	\begin{align*}
		p(x_1, x_2)
			&= \frac{1}{2 \pi} \exp\left( -\frac{\Vert \bA^{-1} \bx \Vert^2}{2} \right) \vert \det \bA^{-1} \vert
	\intertext{[Aufgrund der OrthogonalitûÊt gilt $\bA^{-1} = \bA^\top$.]}
			&= \frac{1}{2 \pi} \exp\left( -\frac{\Vert \bA^\top \bx \Vert^2}{2} \right) \vert \det \bA^\top \vert
	\intertext{[Wieder aufgrund der OrthogonalitûÊt gilt $\Vert \bA^\top \bx \Vert = \Vert \bx \Vert$ und $\det \bA^\top = 1$.]}
			&= \frac{1}{2 \pi} \exp\left( -\frac{\Vert \bx \Vert^2}{2} \right)
	\end{align*}
\end{frame}


% Warum Normalverteilungen verboten sind! (Forts.)
\begin{frame}{Warum Normalverteilungen verboten sind! (Forts.)}{}
	\begin{itemize}
		\item Wir sehen also, dass die Dichte der Quellsignale und der gemischten Signale identisch ist.
		\item Damit haben wir keine Chance, die Mixing Matrix aus den Daten zu schûÊtzen.
	\end{itemize}
\end{frame}

\usefonttheme{professionalfonts}

% disable navigation symbols

% customize table of contents

\setbeamertemplate{subsection in toc}[circle]

\contourlength{0.3pt}

	\mathchoice
      	{\text{\contour{.}{$\displaystyle\oldsum$}}}
      	{\text{\contour{.}{$\textstyle\oldsum$}}}
      	{\text{\contour{.}{$\scriptstyle\oldsum$}}}
      	{\text{\contour{.}{$\scriptscriptstyle\oldsum$}}}
}}

\let\oldprod\prod
\renewcommand{\prod}{\mathop{%

      	{\text{\contour{.}{$\scriptscriptstyle\oldprod$}}}
}}

% commands

% counter
% ====================================================

% define counter
\newcounter{mycounter}
% reset counter for each section
\counterwithin{mycounter}{section}
% print counter
\newcommand{\printmycounter}{%
    \refstepcounter{mycounter}%
    \themycounter
}

% boxes
% ====================================================

% common box styles
\tcbset{common/.style={
	breakable,
	enhanced,
	sharp corners,
	boxrule=0mm,
	boxsep=9pt,
	left=0pt, right=0pt,
	top=0pt, bottom=0pt,
	parbox=false,
	drop small lifted shadow
}}

% boxes for definitions
\tcbset{defbox/.style={
	common,
	colback=lightyellow,
	borderline west={2pt}{0pt}{darkyellow},
}}

% boxes for theorems
\tcbset{theorembox/.style={
	common,
	colback=lightblue,
	borderline west={2pt}{0pt}{darkblue},
}}

% boxes for tasks
\tcbset{taskbox/.style={
	common,
	colback=lightgreen,
	borderline west={2pt}{0pt}{darkgreen},
}}

\newtcolorbox{boxGray}{colback=lightgray!30!white,colframe=white}

% custom environments
% ====================================================

% definition
\NewDocumentEnvironment{mydef}{o}{%
	\begin{tcolorbox}[defbox]
	\textbf{\printmycounter\IfNoValueTF{#1}{}{\label{#1}} Definition:}
}
{%
	\end{tcolorbox}
}

% theorem
\NewDocumentEnvironment{mytheo}{o}{%
	\begin{tcolorbox}[theorembox]
	\textbf{\printmycounter\IfNoValueTF{#1}{}{\label{#1}} Satz:}
}{%
	\end{tcolorbox}
}

% task
\NewDocumentEnvironment{mytask}{o}{%
	\begin{tcolorbox}[taskbox]
	\textbf{\printmycounter\IfNoValueTF{#1}{}{\label{#1}} Aufgabe:}
}{%
	\end{tcolorbox}
}

% proof
\NewDocumentEnvironment{myproof}{}{%
	\footnotesize
	\color{violet}
	\textbf{Beweis:}
}{\hfill $\bm{\square}$}

% algorithm
\NewDocumentEnvironment{myalg}{o}{%
	\begin{tcolorbox}[title=\printmycounter~Algorithmus,fonttitle=\bfseries]
}{%
	\end{tcolorbox}
}

% slide architecture
% ====================================================

% divide frame into two parts
\newcommand{\divideTwo}[4]{
	\begin{minipage}{#1\textwidth}
		#2
	\end{minipage}
	\hfill
	\begin{minipage}{#3\textwidth}
		#4
	\end{minipage}
}

% divide frame into two parts (start on top)
\newcommand{\divideTwoTop}[4]{
	\begin{minipage}[t]{#1\textwidth}
		#2
	\end{minipage}
	\hfill
	\begin{minipage}[t]{#3\textwidth}
		#4
	\end{minipage}
}

% divide frame into three parts
\newcommand{\divideThree}[6]{
	\begin{minipage}{#1\textwidth}
		#2
	\end{minipage}
	\hfill
	\begin{minipage}{#3\textwidth}
		#4
	\end{minipage}
	\hfill
	\begin{minipage}{#5\textwidth}
		#6
	\end{minipage}
}

% divide frame into three parts (start on top)
\newcommand{\divideThreeTop}[6]{
	\begin{minipage}[t]{#1\textwidth}
		#2
	\end{minipage}
	\hfill
	\begin{minipage}[t]{#3\textwidth}
		#4
	\end{minipage}
	\hfill
	\begin{minipage}[t]{#5\textwidth}
		#6
	\end{minipage}
}

% general commands

% keyword
\newcommand{\keyword}[1]{{\usebeamercolor[fg]{structure} \textbf{#1}}}

\newcommand{\rem}[1]{\footnotesize \text{#1}}

% color of eqref references
\let\oldeqref\eqref
\renewcommand{\eqref}[1]{\textcolor{blue}{\oldeqref{#1}}}

% color of citations
\let\oldcite\cite
\renewcommand{\cite}[1]{\textcolor{green}{\oldcite{#1}}}

	\begin{frame}{Kapitel}{}
		\tableofcontents[currentsection,currentsubsection,sectionstyle=show/hide,subsectionstyle=show/show/hide]
	\end{frame}

% commands

% define counter
\newcounter{mycounter}
% reset counter for each section
\counterwithin{mycounter}{section}
% print counter
\newcommand{\printmycounter}{%
    \refstepcounter{mycounter}%
    \themycounter
}

% theorem
\NewDocumentEnvironment{theo}{o}{%
	\textbf{\printmycounter\IfNoValueTF{#1}{}{\label{#1}} Satz:}
}{}

% proof
\NewDocumentEnvironment{prf}{}{%
	\footnotesize
	\color{violet}
	\textbf{Beweis:}
}{\hfill $\bm{\square}$}
